테이블이름 공지사항(Notice)
green_kurly_notice_table

w_no         INT          글번호  
w_type       VARCHAR(20)     공지유형
w_subject    VARCHAR(250)    제목
w_content    text            글내용
w_name       VARCHAR(50)     작성자
w_id         VARCHAR(50)     아이디
w_hit        NUMBER          조회수
w_date       DATE            작성일


$SQL = "CREATE TABLE green_kurly_notice_table (
    idx            INT               NOT NULL AUTO_INCREMENT,
    wNo            INT               NOT NULL,
    wType          VARCHAR(16)       NOT NULL,
    wSubject       VARCHAR(250)      NOT NULL,
    wContent       TEXT              NOT NULL,
    wName          VARCHAR(30)       NOT NULL,
    wId            VARCHAR(16)       NOT NULL,
    wHit           INT               NOT NULL,    
    wDate          timestamp DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY(idx)
) ENGINE=MyISAM DEFAULT CHARSET=utf8mb4";
